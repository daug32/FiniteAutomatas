<while> -> 
        while <condition> do <operator>
<operator> -> 
        <while op>,
	    <assign op>,
	    begin <operator> end
<assign op> ->
        <identifier> = <expression>
<expression> -> 
        <identifier>,
		<expression>+<expression>,
		<expression>-<expression>,
		<expression>*<expression>,
		<expression>/<expression>,
		<expression>%<expression>,
		(<expression>)
<identifier> -> 
        <letter><identifier>
<identifier`> -> 
        <letter><identifier>,
        <number><identifier>
<letter> -> [a-zA-Z_]
<number> -> [0-9]
